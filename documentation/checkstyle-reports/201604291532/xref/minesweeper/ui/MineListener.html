24  package minesweeper.ui;
25  
26  import java.awt.Color;
27  import java.awt.event.ActionEvent;
28  import java.awt.event.ActionListener;
29  import java.awt.event.MouseAdapter;
30  import java.awt.event.MouseEvent;
31  import java.awt.event.MouseListener;
32  import javax.swing.JButton;
33  import javax.swing.SwingUtilities;
34  import minesweeper.logic.Game;
35  import minesweeper.logic.Square;
36  
37  /**
38   *
39   * @author veeti "walther" haapsamo
40   */
41  class MineListener extends MouseAdapter {
42  
43      private final Game game;
44      private final JButton[] buttons;
45      private final int height;
46      private final int width;
47      private final JButton button;
48      private Square square;
49  
50      public MineListener(Game game, JButton[] buttons, JButton button, Square square) {
51          this.game = game;
52          this.buttons = buttons;
53          this.button = button;
54          this.square = square;
55          this.width = game.width;
56          this.height = game.height;
57      }
58  
59      boolean pressed;
60  
61      @Override
62      public void mousePressed(MouseEvent e) {
63          button.getModel().setArmed(true);
64          button.getModel().setPressed(true);
65          pressed = true;
66      }
67  
68      @Override
69      public void mouseReleased(MouseEvent e) {
70          button.getModel().setArmed(false);
71          button.getModel().setPressed(false);
72  
73          if (pressed) {
74              if (SwingUtilities.isRightMouseButton(e)) {
75                  square.toggleFlag();
76                  updateButtons();
77              } else {
78                  step(e);
79              }
80          }
81          pressed = false;
82  
83      }
84  
85      @Override
86      public void mouseExited(MouseEvent e) {
87          pressed = false;
88      }
89  
90      @Override
91      public void mouseEntered(MouseEvent e) {
92          pressed = true;
93      }
94  
95      private void step(MouseEvent e) {
96          // Step into the coordinate of button
97          int index = java.util.Arrays.asList(buttons).indexOf(e.getSource());
98          int x = index / width;
99          int y = index % width;
100 
101         game.turn(x, y);
102         updateButtons();
103     }
104 
105     private void updateButtons() {
106         // Update all buttons
107         for (int i = 0; i < width * height; i++) {
108             int allY = i / width;
109             int allX = i % width;
110 
111             Square currentSquare = game.board.getSquare(allY, allX);
112             JButton currentButton = buttons[i];
113 
114             // Disable clicking on visible / seen squares
115             if (currentSquare.isVisible()) {
116                 currentButton.setBackground(new Color(27, 28, 22));
117             }
118             // No text on empty squares
119             if (currentSquare.isEmpty() && currentSquare.isVisible() && !currentSquare.isFlagged()) {
120                 currentButton.setText(" ");
121             }
122             // F on flagged squares
123             if (currentSquare.isFlagged() && !currentSquare.isVisible()) {
124                 currentButton.setText("F");
125             }
126             // Set number on number square buttons
127             if (currentSquare.isNumber() && currentSquare.isVisible()) {
128                 currentButton.setText(currentSquare.toString());
129                 int value = currentSquare.getValue();
130                 switch (value) {
131                     case 1:
132                         currentButton.setForeground(new Color(0x4eb4fa));
133                         break;
134                     case 2:
135                         currentButton.setForeground(new Color(0xfd971f));
136                         break;
137                     case 3:
138                         currentButton.setForeground(new Color(0xa6e22e));
139                         break;
140                     case 4:
141                         currentButton.setForeground(new Color(0xae81ff));
142                         break;
143                     case 5:
144                         currentButton.setForeground(new Color(0xf6f080));
145                         break;
146                     case 6:
147                         currentButton.setForeground(new Color(0x575855));
148                         break;
149                     case 7:
150                         currentButton.setForeground(new Color(0x000000));
151                         break;
152                     case 8:
153                         currentButton.setForeground(new Color(0xf92672));
154                         break;
155                     default:
156                         break;
157                 }
158 
159             }
160             // If we've lost, turn stepped square red, show all mines, and disable all buttons
161             if (!game.playing && !game.won && currentSquare.isMine()) {
162                 if (currentSquare.isVisible()) {
163                     currentButton.setBackground(new Color(0xf92672));
164 
165                 }
166                 currentButton.setText("*");
167                 for (JButton btn : buttons) {
168                     for (MouseListener l : btn.getMouseListeners()) {
169                         btn.removeMouseListener(l);
170                     }
171                 }
172             }
173 
174         }
175     }
176 
177 }
